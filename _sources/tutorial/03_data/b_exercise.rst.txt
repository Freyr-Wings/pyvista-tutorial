    # mesh points

        [
            [0, 0, 0],

            [1, 1, 0],
            [0, 1, 0],
            [0, 0, 1],

    # mesh faces
    faces = np.hstack(

            [4, 1, 2, 6, 5],  # square

            [4, 3, 0, 4, 7],  # square
        ]
    )

    # plot each face with a different color
    surf.plot(
